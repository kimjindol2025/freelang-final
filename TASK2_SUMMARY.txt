💻 사용 예시
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

// 기본 사용
new Promise((resolve) => {
  resolve(42);
}).then((v) => println(v));          // 42

// 에러 처리
new Promise((_, reject) => {
  reject("error");
}).catch((e) => println(e));          // error

// 체이닝
Promise.resolve(1)
  .then(v => v + 1)                   // 2
  .then(v => v * 2)                   // 4
  .then(v => println(v));             // 4

// 병렬 작업
Promise.all([
  Promise.resolve(1),
  Promise.resolve(2)
]).then(([a, b]) => println(a + b));  // 3
